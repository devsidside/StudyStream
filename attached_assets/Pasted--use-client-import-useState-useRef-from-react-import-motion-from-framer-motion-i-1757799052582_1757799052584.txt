// India Map Component
function IndiaMap() {
  const [hoveredCity, setHoveredCity] = useState<string | null>(null);
  
  const cities = [
    { lat: 28.6139, lng: 77.2090, name: "Delhi", serviceCount: 2.3 },
    { lat: 19.0760, lng: 72.8777, name: "Mumbai", serviceCount: 1.8 },
    { lat: 12.9716, lng: 77.5946, name: "Bangalore", serviceCount: 2.1 },
    { lat: 13.0827, lng: 80.2707, name: "Chennai", serviceCount: 1.5 },
    { lat: 18.5204, lng: 73.8567, name: "Pune", serviceCount: 1.2 },
    { lat: 17.3850, lng: 78.4867, name: "Hyderabad", serviceCount: 0.9 },
  ];

  const map = new DottedMap({ height: 60, grid: "diagonal" });
  const svgMap = map.getSVG({
    radius: 0.22,
    color: "#00000040",
    shape: "circle",
    backgroundColor: "#f8fafc",
  });

  const projectPoint = (lat: number, lng: number) => {
    const x = ((lng - 68) / (97 - 68)) * 800;
    const y = ((37 - lat) / (37 - 8)) * 400;
    return { x, y };
  };

  return (
    <div className="w-full aspect-[2/1] bg-gradient-to-br from-slate-50 to-slate-100 rounded-lg relative overflow-hidden">
      <Image
        src={`data:image/svg+xml;utf8,${encodeURIComponent(svgMap)}`}
        className="h-full w-full [mask-image:linear-gradient(to_bottom,transparent,white_10%,white_90%,transparent)] pointer-events-none select-none"
        alt="India map"
        height="400"
        width="800"
        draggable={false}
      />
      
      <svg viewBox="0 0 800 400" className="w-full h-full absolute inset-0">
        {cities.map((city, i) => {
          const point = projectPoint(city.lat, city.lng);
          return (
            <g key={`city-${i}`}>
              <circle
                cx={point.x}
                cy={point.y}
                r="4"
                fill="#0BC5EA"
                className="cursor-pointer"
                onMouseEnter={() => setHoveredCity(city.name)}
                onMouseLeave={() => setHoveredCity(null)}
              />
              <circle cx={point.x} cy={point.y} r="4" fill="#0BC5EA" opacity="0.6">
                <animate attributeName="r" from="4" to="15" dur="2s" begin={`${i * 0.3}s`} repeatCount="indefinite" />
                <animate attributeName="opacity" from="0.6" to="0" dur="2s" begin={`${i * 0.3}s`} repeatCount="indefinite" />
              </circle>
              <text x={point.x} y={point.y - 18} textAnchor="middle" className="fill-slate-700 text-xs font-medium">
                {city.name}
              </text>
              <text x={point.x} y={point.y - 5} textAnchor="middle" className="fill-slate-500 text-[10px]">
                {city.serviceCount}k
              </text>
            </g>
          );
        })}
      </svg>
    </div>
  );
}
